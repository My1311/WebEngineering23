3. div a:link[href*='h-brs']

1. 023
o 0 = keinen ID-Selektor
o 2 = 1 Klasse + 1 Pseudo-Klasse
o 3 = 3 Typselektoren, nämlich dreimal div

2. 012
o 0 = keinen ID-Selektor
o 2 = 1 Klasse
o 2 = 2 Elementen

3. 022
o 0 = keinen ID-Selektor
o 2 = 2 Klasse
o 2 = 2 Elementen (div und a)

4. 013
o 0 = keinen ID-Selektor
o 2 = 1 Klasse (:hover)
o 3 = 3 Elementen( nav, a, ::bevor)

5. 112
o 1 = ID-Selektor(#primary-nav)
o 2 = 1 Klasse
o 2 = 2 Elementen(ul, li)

1. #ffffff

2. #000

3. #ab0978

4. rgb(127,255,33)

5. rgba(255,127,33,0.8)

rgba(255,127,33,0.8) = hsla(25, 100%, 56%, 0.8)
